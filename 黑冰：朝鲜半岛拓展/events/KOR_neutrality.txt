country_event = {
	id = KOR_neutrality.2
	title = KOR_neutrality.2.t
	desc = KOR_neutrality.2.d
	picture = GFX_report_event_generic_battle
	is_triggered_only = yes
	#选择爆发内战
	option = {
		name = KOR_neutrality.2.a
		trigger = { tag = KOR }
		start_civil_war = {
			#朝鲜殖民地 = {
				set_cosmetic_tag = KMTKOR_fascism #更名为“日本帝国朝鲜殖民地”
				ruling_party = neutrality #原国家保留的执政党（内战爆发后政府的执政党转变）
				ideology = fascism #发动叛乱的意识形态（ROOT即当前政府的意识形态）
				size = civil_war_size_var #叛乱规模修正（0–1 的比例值），影响装备库存（stockpile）、陆军、空军、海军
				army_ratio = 1.0 #叛军能够保留的陆军规模
				navy_ratio = 0.5 #叛军能够保留的海军规模
				air_ratio = 0.5 #叛军能够保留的空军规模
				keep_unit_leaders = { 202619940708 } #指定指挥官ID，这些指挥官将留在原国家，不会加入叛军
				set_capital = { state = 527 } #设置首都:平壤
				states = { 527 815 816 751 819 820 } #划归给叛军的州
				add_ideas = KOR_opium_plantation #朝鲜鸦片种植地
				add_ideas = KOR_bank #朝鲜银行
			#}
			KOR = {
				set_cosmetic_tag = KOR_CHI_aligned #更名为“大韩民国临时政府”
				set_victory_points = { province = 7125 value = 5 } #京城
				set_victory_points = { province = 7221 value = 15 } #仁川
				set_country_flag = SOV_armor #加载苏联坦克科研
				set_country_flag = USA_air #加载美国飞机科研
				remove_ideas = KOR_opium_plantation #移除：朝鲜鸦片种植地
				remove_ideas = KOR_bank #移除：朝鲜银行
				#池青天
				add_field_marshal_role = {
					Character = KOR_Ji_Cheong_cheon 
					traits = { trait_reckless }
					skill = 2
					attack_skill = 2
					defense_skill = 1
					planning_skill = 1
					logistics_skill = 2
					legacy_id = 2026020801
				}
				#李范奭
				add_corps_commander_role = {
					Character = KOR_Lee_Beom_seok
					traits = { skilled_staffer }
					skill = 1
					attack_skill = 1
					defense_skill = 1
					planning_skill = 2
					logistics_skill = 1
					legacy_id = 2026020802
				}
			}
		}
		ai_chance = { factor = 100 }
		hidden_effect = {
			play_song = "GF_06"
		}
	}
	#选择不爆发内战
	option = {
		name = KOR_neutrality.2.b
		trigger = {
			tag = KOR
		}
		ai_chance = { factor = 0 }
	}
}

#KOR请求CHI允许在杭州建立流亡政府
country_event = {
	id = KOR_neutrality.3
	title = KOR_neutrality.3.t
	desc = KOR_neutrality.3.d
	picture = GFX_report_event_generic_battle
	fire_only_once = yes
	is_triggered_only = yes #本事件由KOR丧失全部核心领土后触发，对应文件common\on_actions\zzz_kor_exile_patch.txt
	#CHI同意
	option = {
		name = KOR_neutrality.3.a
		trigger = { tag = CHI }
		KOR = {
			set_country_flag = KOR_is_government_in_exile #标记进入流亡状态
			set_variable = { KOR.host_nation = CHI } #标记CHI是寄主国
			set_capital = { state = 596 } #杭州
			set_victory_points = { province = 7191 value = 100 } #杭州
		}
		CHI = {
			596 = { transfer_state_to = KOR } #移交领土杭州给KOR
		}
		news_event = { id = KOR_neutrality.4 } #蒋介石接受了大韩民国临时政府的请求
		ai_chance = {
			base = 100
		}
	}
	#CHI拒绝
	option = {
		name = KOR_neutrality.3.b
		trigger = { tag = CHI }
		ai_chance = {
			base = 0
		}
	}
}
