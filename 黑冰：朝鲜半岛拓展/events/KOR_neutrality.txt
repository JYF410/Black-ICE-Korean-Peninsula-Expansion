#KOR主动进入流亡状态
country_event = {
	id = KOR_neutrality.3
	title = KOR_neutrality.3.t
	desc = KOR_neutrality.3.d
	picture = GFX_report_event_generic_battle
	fire_only_once = yes
	#is_triggered_only = yes #本事件自然触发/条件触发/不依靠固定事件触发
	#AI决策的“隐藏计算层”
	immediate = {
		hidden_effect = {
			set_variable = { var_opinion = ROOT.opinion@FROM} #读取ROOT对FROM的好感度
			divide_variable = { var_opinion = 100 } #从-100 ~ +100变成：-1 ~ +1
			add_to_variable = { var_opinion = 2 } #再平移到：1 ~ 3，最终得到一个“关系系数”变量
		}
	}
	trigger = {
		tag = KOR
		KOR = {
			surrender_progress > 0.2 #投降进度>20%
			has_country_flag = KOR_neutrality_civilwar_started #大韩民国临时政府要员撤离行动
			NOT = { is_in_faction_with = JAP }
			NOT = { has_country_flag = KOR_neutrality_exile_established }
			NOT = { any_controlled_state = { is_core_of = KOR } }
		}
		CHI = {
			exists = yes
			has_capitulated = no
			has_full_control_of_state = 596 #杭州
		}
	}
	#CHI同意接收流亡政府（核心）
	option = {
		name = KOR_neutrality.3.a
		set_country_flag = KOR_neutrality_exile_established
		596 = { transfer_state_to = KOR } #临时领土：杭州
		set_capital = { state = 596 }
		set_victory_points = { province = 7191 value = 100 } #杭州
		add_ideas = KOR_CHI_political_reform_hangzhou #大韩民国临时政府驻杭州馆
		set_country_flag = KOR_is_government_in_exile #标记进入流亡状态
		set_variable = { KOR.host_nation = CHI } #标记CHI是寄主国
		if = {
			limit = {
				CHI = {
					is_in_faction = yes
					is_faction_leader = yes
				}
			}
			CHI = { add_to_faction = KOR }
		}
		FROM = {
			country_event = {
				id = BBA_italo_ethiopian_war_events.101 #告诉ETH：你被接收了
				hours = 8
				random_hours = 8
			}
		}
		ai_chance = { factor = 99 }
	}
	#选择放弃抵抗
	option = {
		name = KOR_neutrality.3.b
		trigger = {
			tag = KOR
		}
		ai_chance = { factor = 1 }
	}
}
