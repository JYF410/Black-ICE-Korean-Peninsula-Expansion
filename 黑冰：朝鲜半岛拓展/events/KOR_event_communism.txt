#斯大林接受了我们的北萨哈林请求（由KOR触发）
country_event = {
	id = KOR_communism.7
	title = KOR_communism.7.t
	desc = KOR_communism.7.d
	picture = GFX_report_event_stalin_02
	is_triggered_only = yes
	option = {
		name = KOR_communism.7.a
		trigger = { tag = KOR }
		KOR = {
			set_country_flag = KOR_sov_allow_us_get_Sakhalin_flag #斯大林允许朝鲜军队驻扎在北萨哈林
			transfer_state = 655
			if = {
				limit = {
					537 = {
						is_owned_by = SOV
					}
				}
				transfer_state = 537
			}
			hidden_effect = {
				set_capital = { state = 655 } #北萨哈林
				set_victory_points = { province = 1742 value = 50 }
				set_victory_points = { province = 1755 value = 10 }
				set_victory_points = { province = 855 value = 5 }
				655 = {
					add_core_of = KOR
					remove_core_of = SOV
				}
			}
			diplomatic_relation = {
				country = SOV
				relation = non_aggression_pact #签订互不侵犯条约
				active = yes
			}
			set_rule = {
				can_join_factions = no #规则：禁止加入阵营
			}
			add_ideas = KOR_ai_always_NOT_join_ally #AI修正：不会主动加入盟友战争
			add_ideas = KOR_ai_always_NOT_call_ally #AI修正：不会主动请求盟友加入目前的战争
		}
		hidden_effect = {
			play_song = "GF_122" #播放bgm：大东北是我的家乡
		}
	}
}
