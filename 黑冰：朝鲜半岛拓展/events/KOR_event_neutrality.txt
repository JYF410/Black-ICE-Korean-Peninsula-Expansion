	title = KOR_neutrality.2.t
	desc = KOR_neutrality.2.d
	picture = GFX_report_event_portnavy_mutiny1
	is_triggered_only = yes
	#选择爆发内战
	option = {
		name = KOR_neutrality.2.a
		trigger = { tag = KOR }
		start_civil_war = {
			#朝鲜殖民地 = {
				set_cosmetic_tag = KMTKOR_fascism #更名为“日本帝国朝鲜殖民地”
				ruling_party = neutrality #原国家保留的执政党（内战爆发后政府的执政党转变）
				ideology = fascism #发动叛乱的意识形态
				size = civil_war_size_var #叛乱规模修正（0–1 的比例值），影响装备库存（stockpile）、陆军、空军、海军
				army_ratio = 1.0 #朝鲜殖民地能够保留的陆军规模
				navy_ratio = 0 #朝鲜殖民地能够保留的海军规模
				air_ratio = 0 #朝鲜殖民地能够保留的空军规模
				set_capital = { state = 527 } #设置首都:平壤
				states = { 527 815 816 751 819 820 } #划归给朝鲜殖民地的州
				hidden_effect = {
					add_ideas = kantogun_administration #关东军司令部
					add_ideas = KOR_opium_plantation #朝鲜鸦片种植地
					add_ideas = KOR_bank #朝鲜银行
				}
			#}
			KOR = {
				set_cosmetic_tag = KOR_CHI_aligned #更名为“大韩民国临时政府”
				hidden_effect = {
					set_victory_points = { province = 7125 value = 5 } #京城
					set_victory_points = { province = 7221 value = 15 } #仁川
					set_country_flag = SOV_armor #加载苏联坦克科研
					set_country_flag = USA_air #加载美国飞机科研
					add_ideas = foreign_cooperative #合作主义
					add_ideas = labor_allowed #允许女性工作
					remove_ideas = kantogun_administration #关东军司令部
					remove_ideas = KOR_opium_plantation #移除：朝鲜鸦片种植地
					remove_ideas = KOR_bank #移除：朝鲜银行
				}
			}
		}
		ai_chance = { factor = 100 }
		hidden_effect = {
			play_song = "GF_06" #播放bgm：牵丝戏
		}
	}
	#选择不爆发内战
	option = {
		name = KOR_neutrality.2.b
		trigger = { tag = KOR }
		ai_chance = { factor = 0 }
	}
}

	id = KOR_neutrality.4
	title = KOR_neutrality.4.t
	desc = KOR_neutrality.4.d
	picture = GFX_KOR_KMT_allow_korea_event
	fire_only_once = yes #只能触发一次
	is_triggered_only = yes
	option = {
		name = KOR_neutrality.4.a
		trigger = { tag = KOR }
		CHI = {
			596 = { transfer_state_to = KOR } #移交领土杭州给KOR
		}
		KOR = {
			set_country_flag = KOR_neutrality_host_country_decision_completed #对应文件：common\decisions\korea.txt，去除杭州抵抗度
			set_variable = { KOR.host_nation = CHI } #标记CHI是寄主国
			hidden_effect = {
				set_capital = { state = 596 } #迁都杭州
				set_victory_points = { province = 7191 value = 50 } #杭州
				set_victory_points = { province = 4042 value = 15 } #钱塘
			}
		}
		add_ideas = KOR_CHI_political_reform_hangzhou #大韩民国临时政府驻杭州领事馆
		hidden_effect = {
			CHI = {
				remove_opinion_modifier = {
					target = KOR
					modifier = holds_our_cores #他们有我们的核心省份
					modifier = claims_on_us #他们对我们的土地有宣称权！
				}
			}
		}
		ai_chance = {
			base = 500
		}
	}
	option = {
		name = KOR_neutrality.3.b
		trigger = { tag = JAP }
		JAP = {
			add_stability = -0.01 #稳定度-1%
			add_war_support = -0.01 #战争支持度-1%
		}
		ai_chance = {
			base = 100
		}
	}
}
