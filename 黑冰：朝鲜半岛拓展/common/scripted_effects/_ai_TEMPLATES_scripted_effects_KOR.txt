#一旦在该文件夹中编写了模板，并且该国家处于“ai_update_country_logic”状态，进入游戏后加载30天，就可以从AI外交界面访问交互UI。

ai_militia_templates/KOR = {
	ai_militia_templates = yes #开头和结尾都要有
	clear_temp_array = num_valid_templates
	#这些可选的阵列将使ai在解锁它们的技术研究完成后将单位添加到模板中
	#重要的是，下面的模板实际上不是从这些单元开始的
	#add_to_temp_array = { either_support = token:anti_air } #例如ai一旦研究了此防空科技，它就会自动添加到这个模板中
	#add_to_temp_array = { either_regiment_x = 3 } #兵团单位的x坐标，支援单位总是被放置在x=0,这里的x=3是指新研究的营会放在编制的第3列
	add_to_temp_array = { num_valid_templates = 1 } 
	#朝鲜义勇军
	if = {
		limit = {
			NOT = {
				has_country_flag = ai_militia_template
				has_government = fascism
			}
			if = {
				limit = { check_variable = { ai_template_skip = 1 } }
				template_bypass = yes
			}
			else = {
				has_war = no #这些触发器使ai增加解锁单位所需的技术的研究分数。如果ai没有在启动时解锁设备，应该经常检查它们。
			}
		}
		set_temp_variable = { ai_has_added_template = 1 }
		set_country_flag = { flag = ai_militia_template value = 1 }
		division_template = {
			name = "KOR Militia Division I"	 #所有模板都应该使用角色特定的名称，而不是国家特定的名称
			priority = @INFANTRY_PRIO
			regiments = {
				militia = { x = 0 y = 0 }
				militia = { x = 0 y = 1 }
				militia = { x = 0 y = 2 }
				militia = { x = 1 y = 0 }
				militia = { x = 1 y = 1 }
				militia = { x = 2 y = 0 }
				militia = { x = 2 y = 1 }
			}
			support = {
				DIV_HQ = { x = 0 y = 0 }
				recon = { x = 0 y = 1 }
				engineer = { x = 0 y = 2 }
			}
		}
	}
	#二阶民兵师
	else_if = {
		limit = {
			has_country_flag = {
				flag = ai_militia_template value = 1
			}
			if = {
				limit = { check_variable = { ai_template_skip = 1 } }  #一次性效果，检查ai_template_skip是否为0
				template_bypass = yes
			}
			else = {
				ai_has_engineer = yes  #必须先解锁工兵连科技，这将检查前一个模板中的所有either_support或either_regiment步骤是否已经完成
			}
		}
		set_temp_variable = { ai_has_added_template = 1 }
		set_country_flag = { flag = ai_militia_template value = 2 }
		division_template = {
			name = "KOR Militia Division II"	
			priority = @MILITIA_PRIO
			regiments = {
				militia = { x = 0 y = 0 }
				militia = { x = 0 y = 1 }
				militia = { x = 0 y = 2 }
				militia = { x = 0 y = 3 }

				militia = { x = 1 y = 0 }
				militia = { x = 1 y = 1 }
				militia = { x = 1 y = 2 }

				militia = { x = 2 y = 0 }
				militia = { x = 2 y = 1 }
				militia = { x = 2 y = 2 }

				mountain_artillery_brigade = { x = 3 y = 0 }
				mountain_artillery_brigade = { x = 3 y = 1 }
			}
			support = {
				DIV_HQ = { x = 0 y = 0 }
				recon = { x = 0 y = 1 }
				engineer  = { x = 0 y = 2 }
			}
		}
	}
	ai_militia_templates = yes #开头和结尾都要有
}
