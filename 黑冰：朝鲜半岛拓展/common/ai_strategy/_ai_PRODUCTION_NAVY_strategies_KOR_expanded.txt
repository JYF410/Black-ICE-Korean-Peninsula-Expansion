#AI生产舰船策略（第一层）
#unit_ratio：生产预期值（不会直接下生产命令，是一个 “我要有多少” 的目标值）
#build_ship：立即生产值（立刻生效，会根据优先级抢占生产船坞，并与其他舰种竞争优先级value）

#KOR航母生产策略
KOR_CV_naval_production = {
	allowed = { tag = KOR }
	enable = {
		num_of_naval_factories > 15
		carrier < 1
	}
	abort = {
		carrier > 2
	}
	ai_strategy = {
		type = unit_ratio
		id = carrier
		value = 5
	}
	ai_strategy = {
		type = build_ship
		id = carrier
		value = 5
	}
}
